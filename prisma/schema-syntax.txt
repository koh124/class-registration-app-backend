generator client {
  provider = "prisma-client-js"
  binaryTargets = ["native", "debian-openssl-1.1.x"]
}

datasource db {
  provider = "mysql"
  url      = env("DATABASE_URL")
}

// 原則モデルはキャメルケース
model University {
  id Int @id @default(autoincrement())
  name String
  courses Courses[] // Prisma独自のシンタックスでhas manyのリレーションを表現する
}

model User {
  id Int @id @default(autoincrement())
  name String?
  userId String @unique
  password String
}

model Courses {
  id Int @id @default(autoincrement())
  name String
  introduction String?
  period String
  date String
  universityId Int

  // belongs toの外部キー参照はPrisma独自のシンタックスで行う
  任意のリレーションの命名 関係モデル名 @relation(fields: [参照元のモデルのキー], references: [参照先の主キー])
  university University @relation(fields: [universityId], references: [id])
}

enum UserCourseStateEnum {
  ENROLLED
  NOT_ENROLLED
  COMPLETED
  NOT_COMPLETED
}

model UserCourseStateMaster {
  id Int @id @default(autoincrement())
  state UserCourseStateEnum
  userCourseStates UserCourseState[]
}

model UserCourseState {
  userId Int
  courseId Int
  stateId Int
  state UserCourseStateMaster @relation(fields: [stateId], references: [id])

  // 複合キーの記述は下記のように行う
  @@id([userId, courseId])
}
